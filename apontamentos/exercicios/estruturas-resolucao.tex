	T(N) =
	\begin{cases}
		0 & N \leq 0 \\
		2 \times T_{altura}(N/2) + 2 \times T(N/2) & N > 0
	\end{cases}
\]

	\begin{cases}
		0 & N \leq 0 \\
		N + 2 \times T(N/2) & N > 0
	\end{cases}
